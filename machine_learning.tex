						\begin{aligned}
							& \underset{\mathbf{x}}{\text{min}}
							& & f(\mathbf{x}) = \mathbb{E}_{\Theta}\left[{f}_{\Theta}(\mathbf{x}) = \mathrm{error}_\mathrm{CV}\right]\\
							& \text{where}
							& & \Theta\mathrm{:realizations}
						\end{aligned}
					\end{equation*}
				\end{exampleblock}
